# 友元函数
> 在类中声明一个函数，用来修改pravite 成员变量   
> 函数的实现在类的外部

```
#include<iostream>
using namespace std;

//友元函数
class A {
	// friend 关键字 可以让p 访问私有变量
	friend void modify_i(A* p, int a);
private:
	int i;
public:
	A(int i) :i(i) {

	}
	void myPrint() {
		cout << i << endl;
	}
};

void modify_i(A* p, int a) {
	p->i = a;
}

void main() {
	A* p = new A(3);
	//p->i = 4; 这里不能直接访问i
	p->myPrint();
	modify_i(p, 5);
	p->myPrint();
	delete p;

	system("pause");
}
```

# 重载

1. 运算符重载
> 语法：T operator<<(T & t)   
> 注意：访问修饰符

```
// 重载+
#include <iostream>
using namespace std;
class Point {
public:
	int x;
	int y;
public:
	Point(int x = 0, int y = 0) {
		this->x = x;
		this->y = y;
	}
	//成员函数，运算符重载
	Point operator+(Point &p2) {
		Point tmp(this->x + p2.x, this->y + p2.y);
		return tmp;
	}
	void myprint() {
		cout << x << "," << y << endl;
	}
};

void main() {
	Point p1(10, 20);
	Point p2(20, 10);

	//运算符的重载，本质还是函数调用
	//p1.operator+(p2)
	Point p3 = p1 + p2;

	p3.myprint();

	system("pause");
}
```

## virtual

1. 虚继承
> 解决多继承中的二义性问题

```

class A {
public:
	char* name;
};

class A1 :  public A {

};

class A2 :  public A {

};

class B : public A1, public A2 {

};

void main() {
	B b;
	//b.name = "jason";// 这里报错 指向不明确
	//指定父类显示调用
	
	b.A1::name = "jason";
	b.A2::name = "jason";
	system("pause");
}
```
如果b 的父类中都存在 name 属性 所以导致指向不明确。
解决方法：在共有类中添加virtual 修饰符

```
class A1 :  virtual public A {

};

class A2 : virtual public A {

};
```

注意： virtual 无法修饰属性
